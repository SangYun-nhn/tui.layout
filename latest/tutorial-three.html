        var cal = new tui.component.Calendar({
        element: $('#cal'),
        year: 1983,     // 달력이 보여줄 날짜-년도
        month: 5,       // 달력이 보여줄 날짜-월
        titleFormat: "yyyy-mm", //설정될 title의 형식
        todayFormat: "yyyy - mm - dd (D)", // 설정된 오늘의 날짜 형식
        monthTitles: ["JAN", "FEB", "MAR", "APR", "MAY", "JUN", "JUL", "AUG", "SEP", "OCT", "NOV", "DEC"], //월의 이름을 설정 "title" 세팅시 적용
        dayTitles: ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'] // 요일 문자값
    });
